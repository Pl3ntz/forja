  }

% Modern-specific commands

% Subheading with intro paragraph rendered between subheading row and bullets
% Usage: \resumeSubheadingWithIntro{company}{date}{role}{location}
%        followed by \resumeIntro{text} then \resumeItemListStart ... \resumeItemListEnd
\newcommand{\resumeIntro}[1]{%
  \par\vspace{2pt}%
  \small{#1}\par\vspace{-4pt}%
}

% Per-experience skills line rendered after bullets
% Usage: \resumeSkillsLine{Python, Spark, ...}
\newcommand{\resumeSkillsLine}[1]{%
  \par\vspace{1pt}%
  \textit{\small{Skills: #1}}\par\vspace{2pt}%
}

% Certification item: name, issuer, year
% Usage: \resumeCertification{name}{issuer}{year}
\newcommand{\resumeCertification}[3]{%
  \item\small{\textbf{#1} --- #2 (#3)\vspace{-2pt}}%
}
